### Query Examples

#### Create Operation
```typescript
// Before
const files = await client.query(
    "INSERT INTO files (user_email, file_name, file_content) VALUES ($1, $2, $3)", 
    [email, Name, fileContent]
);

// After
const file = await prisma.file.create({
    data: {
        userEmail: email,
        fileName: Name,
        fileContent: fileContent,
        isPasswordProtected: isPasswordProtected || false
    }
});
```

#### Read Operation
```typescript
// Before
const files = await client.query(
    "SELECT * FROM files WHERE user_email = $1 AND file_name = $2", 
    [email, Name]
);

// After
const file = await prisma.file.findFirst({
    where: {
        userEmail: email,
        fileName: Name
    }
});
```

#### List Operation
```typescript
// Before
const files = await client.query(
    "SELECT file_name, updated_at FROM files WHERE user_email = $1", 
    [email]
);

// After
const files = await prisma.file.findMany({
    where: {
        userEmail: email
    },
    select: {
        fileName: true,
        updatedAt: true
    }
});
```

#### Delete Operation
```typescript
// Before
await client.query(
    "DELETE FROM files WHERE user_email = $1 AND file_name = $2", 
    [email, Name]
);

// After
const deletedFile = await prisma.file.deleteMany({
    where: {
        userEmail: email,
        fileName: Name
    }
});
```
